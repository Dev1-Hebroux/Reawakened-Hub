def add_title(doc, title, subtitle=None):
    p = doc.add_paragraph()
    run = p.add_run(title)
    run.bold = True
    run.font.size = Pt(20)
    p.alignment = WD_ALIGN_PARAGRAPH.CENTER
    if subtitle:
        p2 = doc.add_paragraph()
        r2 = p2.add_run(subtitle)
        r2.font.size = Pt(12)
        p2.alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph()

def add_day(doc, day_num, title, scripture, devotional_paragraphs, reflection):
    doc.add_heading(f"Day {day_num}: {title}", level=2)
    # Scripture line
    p = doc.add_paragraph()
    r = p.add_run("Scripture: ")
    r.bold = True
    p.add_run(scripture)
    # Devotional label
    p = doc.add_paragraph()
    r = p.add_run("Devotional:")
    r.bold = True
    # Add devotional paragraphs
    for para in devotional_paragraphs:
        doc.add_paragraph(para)
    # Reflection question
    p = doc.add_paragraph()
    r = p.add_run("Reflection Question: ")
    r.bold = True
    p.add_run(reflection)
    doc.add_paragraph()

def sanitize(s):
    # ensure ascii hyphen, avoid exotic dashes
    s = s.replace("—", "-").replace("–", "-").replace("‑", "-")
    return s

from docx import Document
from docx.shared import Pt
from docx.enum.text import WD_ALIGN_PARAGRAPH

def add_title(doc, title, subtitle=None):
    p = doc.add_paragraph()
    run = p.add_run(title)
    run.bold = True
    run.font.size = Pt(20)
    p.alignment = WD_ALIGN_PARAGRAPH.CENTER
    if subtitle:
        p2 = doc.add_paragraph()
        r2 = p2.add_run(subtitle)
        r2.font.size = Pt(12)
        p2.alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph()

def add_day(doc, day_num, title, scripture, devotional_paragraphs, reflection):
    doc.add_heading(f"Day {day_num}: {title}", level=2)
    p = doc.add_paragraph()
    r = p.add_run("Scripture: ")
    r.bold = True
    p.add_run(scripture)
    p = doc.add_paragraph()
    r = p.add_run("Devotional:")
    r.bold = True
    for para in devotional_paragraphs:
        doc.add_paragraph(para.replace("—","-").replace("–","-"))
    p = doc.add_paragraph()
    r = p.add_run("Reflection Question: ")
    r.bold = True
    p.add_run(reflection)
    doc.add_paragraph()
